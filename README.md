# Клонирование репозитория

# Подстановка секретных переменных в окружение (.env)
    DJANGO_SECRET_KEY=
    DJANGO_DEBUG=
    DJANGO_ALLOWED_HOSTS=
    DJANGO_ALLOW_REVERSE=
    DJANGO_INTERNAL_IPS=

# Запуск docker-compose
    docker-compose up -d
